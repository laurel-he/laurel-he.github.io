重启进程脚本

#!/bin/bash

# 此脚本作用是定制化重启服务,假设服务名为myTestServer
# 后台启动后除了make run进程，还会有go run config/grpc_go.yaml等进程，这里假设都能匹配grpc_go.yaml
# 有多个同类型的服务，只是服务名不同
# 手动重启有两种办法：根据时间判断对应的run进程和根据进程信息判断，为避免多个服务同时启动的极端情况，这里根据进程信息来判断

# 1 查询包含make的进程
# 2 查询进程详细信息，找到myTestServer对应的进程
# 3 直接kill
# 4 找到grpc_go.yaml对应的进程
# 5 查看第4步进程的详细信息，找到myTestServer对应的进程
# 6 直接kill
# 7 进入项目目录，pull最新代码
# 8 如果go.mod文件有更新，执行go mod vendor
# 9 nohup make run &后台启动服务
cmd_name="make"
proc_grep_name="grpc_go.yaml"
project_name="myTestServer"
branch_name="feature/5.2/test_cicd"
project_route="/data/myTestServer"
mod_name="web"

# kill make 进程
ps -ef | grep $cmd_name | grep -v grep> tmpMakeProcess.txt
make_process_pid=$(awk '{print $2}' tmpMakeProcess.txt)

for i in $make_process_pid
do
        make_process_detail=$(ls -al /proc/$i | grep $project_name)
        if [[ $make_process_detail =~ $mod_name$ ]];then
                echo "make process id is: "$i", and it will be kill now..."
                sudo kill -9 $i
        fi
done
rm -rf tmpMakeProcess.txt

ps -ef | grep $proc_grep_name | grep -v grep> tmpOtherProcess.txt
proc_process_pid=$(awk '{print $2}' tmpOtherProcess.txt)

for i in $proc_process_pid
do
        proc_process_detail=$(ls -al /proc/$i | grep $project_name)
        if [[ $proc_process_detail =~ $mod_name$ ]];then
                echo "proc process id is: "$i", and it will be kill now..."
                sudo kill -9 $i
        fi
done
rm -rf tmpOtherProcess.txt

cd $project_route"/"$project_name"/"$mod_name
pwd
git pull origin $branch_name
go mod vendor
echo "now it will restart..."
nohup make run &
